function getHoroscope() {
    const input = document.getElementById("zodiacInput").value.trim().toLowerCase();
    const resultDiv = document.getElementById("horoscopeResult");

    if (input in horoscopes) {
        resultDiv.innerHTML = `<p><strong>${capitalizeFirstLetter(input)}:</strong> ${horoscopes[input]}</p>`;
    } else {
        resultDiv.innerHTML = "<p>❌ К сожалению, я не знаю этот знак зодиака. Попробуйте ещё раз.</p>";
    }
}

function capitalizeFirstLetter(string) {
    return string.charAt(0).toUpperCase() + string.slice(1);
}
